44template <class DelimT, class CharT = char, class Traits = std::char_traits<CharT>>
45class ostream_joiner {
46 public:
47 using char_type = CharT;
48 using traits_type = Traits;
49 using ostream_type = std::basic_ostream<CharT, Traits>;
50 using iterator_category = std::output_iterator_tag;
51 using value_type = void;
52 using difference_type = void;
53 using pointer = void;
54 using reference = void;
55
56 private:
57 ostream_type* m_os;
58 DelimT m_delim;
59 bool m_is_first = true;
60
61 public:
62 ostream_joiner(ostream_type& os, const DelimT& delimiter) noexcept(std::is_nothrow_copy_constructible_v<DelimT>)
63 : m_os(std::addressof(os)), m_delim(delimiter) {}
64
65 ostream_joiner(ostream_type& os, DelimT&& delimiter) noexcept(std::is_nothrow_move_constructible_v<DelimT>)
66 : m_os(std::addressof(os)), m_delim(std::move(delimiter)) {}
67
68 template <typename T>
69 ostream_joiner& operator=(const T& value) {
70 if (!m_is_first) {
71 *m_os << m_delim;
72 }
73 m_is_first = false;
74 *m_os << value;
75 return *this;
76 }
